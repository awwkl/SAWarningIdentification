+          // TODO: fix this
+          continue;
+        }
+      } else if (sim instanceof DFRSimilarity) {

+        if (model instanceof BasicModelD || model instanceof BasicModelP) {
+          // score goes NaN for tiny doc lengths
+          // TODO: fix this

+          // TODO: fix this
